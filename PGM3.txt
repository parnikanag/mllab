# Step 1: Import required libraries
import pandas as pd
import matplotlib.pyplot as plt
from sklearn.preprocessing import StandardScaler
from sklearn.decomposition import PCA

# Step 2: Load Iris dataset from local CSV file
file_path = "iris.csv"
df = pd.read_csv(file_path)

# Display dataset information
print("Dataset Shape:", df.shape)
print("\nColumns:\n", df.columns)

# Step 3: Separate features and target
X = df.iloc[:, 0:4].values     # First four columns are features
y = df.iloc[:, 4].values       # Last column is class label

# Step 4: Standardize the features
scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)

# Step 5: Apply PCA (reduce from 4 features to 2 components)
pca = PCA(n_components=2)
X_pca = pca.fit_transform(X_scaled)

# Step 6: Create DataFrame for PCA results
pca_df = pd.DataFrame(
    data=X_pca,
    columns=['Principal Component 1', 'Principal Component 2']
)
pca_df['Species'] = y

# Step 7: Visualize PCA output
plt.figure(figsize=(8, 6))

for species in pca_df['Species'].unique():
    plt.scatter(
        pca_df[pca_df['Species'] == species]['Principal Component 1'],
        pca_df[pca_df['Species'] == species]['Principal Component 2'],
        label=species
    )

plt.xlabel('Principal Component 1')
plt.ylabel('Principal Component 2')
plt.title('PCA of Iris Dataset (4 Features → 2 Components)')
plt.legend()
plt.grid(True)
plt.show()

# Step 8: Display explained variance
print("Explained Variance Ratio:", pca.explained_variance_ratio_)
print("Total Variance Explained:", sum(pca.explained_variance_ratio_))
